78 EditorTypes::EditorType StaticFunctions::findEditorType(const QString fileName)
79 {
80  QFileInfo fileInfo(fileName);
81  QString completeSuffix(fileInfo.completeSuffix().toLower());
82  QString suffix(fileInfo.suffix().toLower());
83  QSettings settings;
84  if (settings.value("FileAssociations/SQLFiles", "").toString().split(",").contains(completeSuffix)
85  || settings.value("FileAssociations/SQLFiles", "").toString().split(",").contains(suffix))
86  return EditorTypes::SQLQuery;
87  return EditorTypes::NoEditor;

372 QString StaticFunctions::password(QString pass, bool encrypt)
373 {
374  QString pas;
375 
376  if (encrypt) {
377  QChar *data;
378  qsrand(QTime::currentTime().second());
379  pas = "*";
380  data = pass.data();
381  while (!data->isNull()) {
382  int seed = (int) qrand() % 9;
383  pas += QString("%1%2").arg(seed).arg(QChar(data->unicode() + seed));
384  ++data;
385  }
386  } else {
387  for (int i = 1; i < pass.length(); i++) {
388  pas += QString("%1").arg(QChar(pass.at(i + 1).unicode() - QString(pass.at(i)).toInt()));
389  i++;
390  }
391  }
392  return pas;
393 }
