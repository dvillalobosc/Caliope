243 void TableMaintenance::itemActivatedSlot(QTreeWidgetItem *item, int column)
244 {
245  if (item->type() == ItemTypes::Database) {
246  QApplication::setOverrideCursor(Qt::WaitCursor);
247  QTreeWidgetItem *itemChild;
248  int counter = 0;
249  emit loadProgress(0);
250  foreach (QTreeWidgetItem *tableItem, tables) {
251  if (tableItem->parent() == item)
252  tableItem->setCheckState(column, item->checkState(column));
253  emit loadProgress((int) (counter * 100 / tables.count()));
254  counter++;
255  }
256  if (item->childCount() == 0) {
257  QStringList databaseTables;
258  if (selectAllLocalTables)
259  databaseTables = serverConnection->database(item->text(0))->getLocalTables();
260  else
261  databaseTables = serverConnection->database(item->text(0))->getTables();
262  counter = 0;
263  foreach (QString table, databaseTables) {
264  itemChild = new QTreeWidgetItem(item, QStringList("`" + item->text(0) + "`.`" + table + "`"), ItemTypes::Table);
265  itemChild->setFlags(Qt::ItemIsUserCheckable | Qt::ItemIsEnabled | Qt::ItemIsSelectable);
266  itemChild->setIcon(0, QIcon(":/images/svg/table.svg"));
267  itemChild->setCheckState(0, Qt::Checked);
268  tables.append(itemChild);
269  emit loadProgress((int) (counter * 100 / databaseTables.count()));
270  }
271  }
272  emit loadProgress(100);
273  QApplication::restoreOverrideCursor();
274  }
275 }
