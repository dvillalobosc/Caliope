145 void ProcessList::reloadData()
146 {
147  switch(qApp->property("DBMSType").toInt()) {
148  case StaticFunctions::MySQL:
149  result = serverConnection->runQuery("SELECT '', `ID`, `USER`, `HOST`, `DB`, `COMMAND`, `TIME`, `STATE`, `INFO` FROM `information_schema`.`PROCESSLIST`");
150  break;
151  case StaticFunctions::MariaDB:
152  result = serverConnection->runQuery("SELECT '', `ID`, `USER`, `HOST`, `DB`, `COMMAND`, `TIME`, `STATE`, `INFO`, `TIME_MS`, `STAGE`, `MAX_STAGE`, `PROGRESS`, `MEMORY_USED`, `EXAMINED_ROWS`, `QUERY_ID` FROM `information_schema`.`PROCESSLIST`");
153  break;
154  case StaticFunctions::PostgreSQL:
155  result = serverConnection->runQuery("SELECT '', datid, datname, procpid, usename, current_query, waiting, xact_start, query_start, backend_start, client_addr, client_port FROM pg_stat_activity");
156  break;
157  case StaticFunctions::Undefined:
158  default:
159  break;
160  }
161  result->takeLast(); //Remove the "Affected rows" line.
162  if (result->count() > 0)
163  processTable->setModelData(result);
164  if (pushButtonStopRefreshing->isChecked())
165  timerRefresh->stop();
166 }
