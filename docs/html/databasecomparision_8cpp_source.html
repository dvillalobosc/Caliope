98 void DatabaseComparision::fillTablesSlot()
99 {
100  QApplication::setOverrideCursor(Qt::WaitCursor);
101  tables.clear();
102  tablesListWidget->setWindowTitle(tr("Original database server: %1").arg(serverConnection->getHostName()));
103  tablesListWidget->setHeaderLabel(tablesListWidget->windowTitle());
104  foreach (QString database, serverConnection->getDatabases()) {
105  QTreeWidgetItem *item = new QTreeWidgetItem((QTreeWidget*)0, QStringList(database), ItemTypes::Database);
106  item->setIcon(0, QIcon(":/images/svg/server-database.svg"));
107  item->setFlags(Qt::ItemIsUserCheckable | Qt::ItemIsEnabled | Qt::ItemIsSelectable);
108  item->setCheckState(0, Qt::Unchecked);
109  tables.append(item);
110  foreach (QString table, serverConnection->database(database)->getTables()) {
111  QTreeWidgetItem *itemChild = new QTreeWidgetItem(item, QStringList(database + "." + table), ItemTypes::Table);
112  itemChild->setFlags(Qt::ItemIsUserCheckable | Qt::ItemIsEnabled | Qt::ItemIsSelectable);
113  itemChild->setIcon(0, QIcon(":/images/svg/table.svg"));
114  itemChild->setCheckState(0, Qt::Unchecked);
115  tables.append(itemChild);
116  }
117  }
118  tablesListWidget->clear();
119  tablesListWidget->insertTopLevelItems(0, tables);
120  QApplication::restoreOverrideCursor();
121 
122  pushButtonPrimaryCompare->setEnabled(false);
123  ConnectDialog *connectFrom = new ConnectDialog(secondaryServerConnection);
124  if (connectFrom->exec() == QDialog::Accepted) {
125  if (!connectFrom->getConnectionPerformed()) {
126  if (secondaryServerConnection->isOpened())
127  secondaryServerConnection->close();
128  secondaryServerConnection->setUserName(connectFrom->getValues().at(3));
129  secondaryServerConnection->setHostName(connectFrom->getValues().at(1));
130  secondaryServerConnection->setPassword(connectFrom->getValues().at(4));
131  secondaryServerConnection->setDatabase(connectFrom->getValues().at(5));
132  secondaryServerConnection->setPort(connectFrom->getValues().at(2).toUInt());
133  if (!secondaryServerConnection->open())
134  QMessageBox::critical(this, tr("Cannot connect to the server"), secondaryServerConnection->lastError());
135  }
136  }
137  if (secondaryServerConnection->isOpened()) {
138  secondaryTablesListWidget->setWindowTitle(tr("Secondary database server: %1").arg(secondaryServerConnection->getHostName()));
139  secondaryTablesListWidget->setHeaderLabel(secondaryTablesListWidget->windowTitle());
140  pushButtonPrimaryCompare->setEnabled(true);
141  emit statusBarMessage(tr("Connected successfully to: %1").arg(secondaryServerConnection->getHostName()));
142  QApplication::setOverrideCursor(Qt::WaitCursor);
143  secondaryTables.clear();
144  foreach (QString database, secondaryServerConnection->getDatabases()) {
145  QTreeWidgetItem *item = new QTreeWidgetItem((QTreeWidget*)0, QStringList(database), ItemTypes::Database);
146  item->setIcon(0, QIcon(":/images/svg/server-database.svg"));
147  item->setFlags(Qt::ItemIsUserCheckable | Qt::ItemIsEnabled | Qt::ItemIsSelectable);
148  item->setCheckState(0, Qt::Unchecked);
149  secondaryTables.append(item);
150  foreach (QString table, secondaryServerConnection->database(database)->getTables()) {
151  QTreeWidgetItem *itemChild = new QTreeWidgetItem(item, QStringList(database + "." + table), ItemTypes::Table);
152  itemChild->setFlags(Qt::ItemIsUserCheckable | Qt::ItemIsEnabled | Qt::ItemIsSelectable);
