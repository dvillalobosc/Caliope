362 QList<QStringList>* DBMS::runQuery(QString queryToExecute, bool addHeaders)
363 {
364  queryExecutionTime.restart();
365  QList<QStringList> *rows = new QList<QStringList>();
366  QStringList row;
367  int status = 0;
368 
369  // qDebug() << queryToExecute;
370 
371  if (isOpened()) {
372  switch(qApp->property("DBMSType").toInt()) {
373  case StaticFunctions::MySQL:
374  case StaticFunctions::MariaDB:
375  if (query(queryToExecute) == 0) {
376  do {
377  /* did current statement return data? */
378  mariadbResults = mysql_store_result(mariadbConnection);
379  if (mariadbResults) {
380  /* yes; process rows and free the result set */
381  if (addHeaders) {
382  while((field = mysql_fetch_field(mariadbResults)))
383  row.append(field->name);
384  rows->append(row);
385  row.clear();
386  }
387  while((record = mysql_fetch_row(mariadbResults))) {
388  for (int counter = 0; counter < (int) mysql_num_fields(mariadbResults); counter++)
389  row.append(QString::fromUtf8(record[counter]));
390  rows->append(row);
391  row.clear();
392  }
393  rows->append(QStringList() << tr("Rows in set: %1. Elapsed time: %2.").arg(mysql_affected_rows(mariadbConnection)).arg(millisecondsToTime(queryExecutionTime.elapsed())));
394  mysql_free_result(mariadbResults);
395  } else {/* no result set or error */
396  if (mysql_field_count(mariadbConnection) == 0) {
397  //rows->append(QStringList() <<tr("Rows in set: %1").arg(mysql_affected_rows(mariadbConnection)));
398  rows->append(QStringList() << tr("Rows in set: %1. Elapsed time: %2.").arg(mysql_affected_rows(mariadbConnection)).arg(millisecondsToTime(queryExecutionTime.elapsed())));
399  } else { /* some error occurred */
400  rows->append(QStringList() << errorOnExecution(lastError()));
401  break;
402  }
403  }
404  /* more results? -1 = no, >0 = error, 0 = yes (keep looping) */
405  if ((status = mysql_next_result(mariadbConnection)) > 0) {
406  rows->append(QStringList() << errorOnExecution(lastError()));
407  //rows->append(QStringList() << tr("Rows in set: %1").arg(0)); //Implement this for PSQL
408  rows->append(QStringList() << tr("Rows in set: %1. Elapsed time: %2 seconds.").arg(0).arg(0)); //Implement this for PSQL
409  }
410  } while (status == 0);
411  } else {
412  rows->append(QStringList() << errorOnExecution(lastError()));
413  //rows->append(QStringList() << tr("Rows in set: %1").arg(0));
414  rows->append(QStringList() << tr("Rows in set: %1. Elapsed time: %2 seconds.").arg(0).arg(0));
415  }
416  break;
417  case StaticFunctions::PostgreSQL:
418 #if USEPOSTGRES == 1
419  query("BEGIN");
420  query("DECLARE mycursor CURSOR FOR " + queryToExecute);
421  query("FETCH ALL IN mycursor");
422  if (PQnfields(postgresqlResults) == 0) {
423  rows->append(QStringList() << errorOnExecution(lastError()));
424  rows->append(QStringList() << tr("Rows in set: %1").arg(0));
425  }
426  if (addHeaders) {
427  for (int counter = 0; counter < PQnfields(postgresqlResults); counter++)
428  row.append(PQfname(postgresqlResults, counter));
429  rows->append(row);
430  row.clear();
431  }
432  for (int counter = 0; counter < PQntuples(postgresqlResults); counter++) {
433  for (int column = 0; column < PQnfields(postgresqlResults); column++)
434  row.append(PQgetvalue(postgresqlResults, counter, column));
435  rows->append(row);
436  row.clear();
437  }
438  rows->append(QStringList() << tr("Rows in set: %1").arg(PQntuples(postgresqlResults)));
439  query("CLOSE mycursor");
440  query("END");
441  PQclear(postgresqlResults);
442 #endif
443  break;
444  case StaticFunctions::Undefined:
445  default:
446  rows->append(QStringList() << tr("Undefined Database driver.") << "-");
447  break;
448  }
449  // for (int counter = 0; counter < rows->count(); counter++)
450  // qDebug() << "*" << rows->at(counter);
451  }
452  return rows;
453 }
