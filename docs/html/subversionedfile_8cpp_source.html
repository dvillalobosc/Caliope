56 QString SubversionedFile::svnCommand(QStringList commands)
57 {
58  QProcess *process = new QProcess(this);
59  process->start("svn", QStringList() << commands << fileName);
60  process->waitForFinished();
61  QByteArray output = process->readAll();
62  QString result;
63  for (int counter = 0; counter < output.size(); ++counter)
64  result += output.at(counter);
65  return result;
66 }
