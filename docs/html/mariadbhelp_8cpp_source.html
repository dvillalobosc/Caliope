88 void MariaDBHelp::fillHelpTreeWidget(QString filter)
89 {
90  helpTreeWidget->clear();
91  QList<QTreeWidgetItem *> items;
92  QTreeWidgetItem *item;
93  QTreeWidgetItem *itemChild;
94  QStringList rows2;
95  rows = serverConnection->runQuery("SELECT `help_category_id`, `name` FROM `mysql`.`help_category` ORDER BY `name`");
96  rows->takeLast(); //Remove the "Affected rows" line.
97  for (int counter = 0; counter < rows->count(); counter++) {
98  item = new QTreeWidgetItem((QTreeWidget*)0, QStringList(rows->at(counter).at(1)));
99  if (filter.isEmpty())
100  rows2 = serverConnection->runQuerySingleColumn("SELECT `name` FROM `mysql`.`help_topic` WHERE `help_category_id` = "
101  + rows->at(counter).at(0) + " ORDER BY `name`");
102  else
103  rows2 = serverConnection->runQuerySingleColumn("SELECT `name` FROM `mysql`.`help_topic` WHERE `help_category_id` = "
104  + rows->at(counter).at(0) + " AND `name` LIKE '%"+ filter
105  + "%' ORDER BY `name`");
106  for (int counter2 = 0; counter2 < rows2.count(); counter2++) {
107  itemChild = new QTreeWidgetItem(item, QStringList(rows2.at(counter2)));
108  itemChild->setIcon(0, QIcon(":/images/svg/server-database.svg"));
109  }
110  item->setIcon(0, QIcon(":/images/svg/server-database.svg"));
111  items.append(item);
112  }
113  helpTreeWidget->insertTopLevelItems(0, items);
114 }

116 void MariaDBHelp::showTopicSlot(QTreeWidgetItem *item, int column)
117 {
118  rows = serverConnection->runQuery("SELECT `description`, `example`, `url` FROM `mysql`.`help_topic` WHERE `name` = '" + item->text(column) + "'");
119  rows->takeLast(); //Remove the "Affected rows" line.
120  if (rows->count() > 0)
121  helpView->setHtml("<H1>" + tr("Syntax:") + "</H1>" + QString(rows->at(0).at(0)).replace("\n", "<BR />")
122  + "<H1>" + tr("Example:") + "</H1>"
123  + rows->at(0).at(1)
124  + "<H1>" + tr("URL:") + "</H1>"
125  + "<A href='" + rows->at(0).at(2) + "'>" + rows->at(0).at(2) + "</A>");
126 }
