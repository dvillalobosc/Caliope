168 void Preferences::fillModelData()
169 {
170  connectionsData->clear();
171  settings.beginGroup("ServerConnections");
172  QStringList connections = settings.allKeys();
173  connections.removeAt(connections.indexOf("StorePassword"));
174  for (int step = 0; step < connections.count(); step++) {
175  QStringList list = StaticFunctions::explodeConnectionString(connections.at(step));
176  connectionsData->append(QStringList() << "" << list.at(0) << connections.at(step) << list.at(1) << list.at(2) << list.at(3) << list.at(4) << list.at(5));
177  }
178  settings.endGroup();
179 }

216 void Preferences::connectioItemChangedSlot(QStandardItem *item)
217 {
218  settings.beginGroup("ServerConnections");
219  if ((item->data(Qt::UserRole).toString() != item->data(Qt::DisplayRole).toString()) && (item->column() == 0))
220  settings.remove(item->data(Qt::UserRole).toString());
221  settings.setValue(listConnections->indexData(item->row(), 1, Qt::EditRole).toString()
222  , QString("%1:%2@%3:%4/%5 %6")
223  .arg(listConnections->indexData(item->row(), 0, Qt::EditRole).toString())
