208 void Preferences::fillModelData()
209 {
210  connectionsData->clear();
211  settings.beginGroup("ServerConnections");
212  QStringList connections = settings.allKeys();
213  connections.removeAt(connections.indexOf("StorePassword"));
214  connections.removeAt(connections.indexOf("SortConnectionList"));
215  foreach (QString connection, connections)
216  connectionsData->append(QStringList() << "" << connection << StaticFunctions::explodeConnectionString(connection));
217  settings.endGroup();
218 }
