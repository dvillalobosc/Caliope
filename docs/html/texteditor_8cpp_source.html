360 void TextEditor::matchBracket()
361 {
362  QList<QTextEdit::ExtraSelection> selections = textEditor->extraSelections();
363  textEditor->setExtraSelections(selections);
364  TextBlockData *data = static_cast<TextBlockData *>(textEditor->textCursor().block().userData());
365  if (data) {
366  QVector<BracketInfo *> infos = data->brackets();
367  int pos = textEditor->textCursor().block().position();
368  for (int i = 0; i < infos.size(); ++i) {
369  foreach (char leftBracket, StaticFunctions::bracketLeftList()) {
370  char rightBracket = StaticFunctions::rightBracketMatch(leftBracket);
371  BracketInfo *info = infos.at(i);
372  int curPos = textEditor->textCursor().position() - textEditor->textCursor().block().position();
373  if (info->position == curPos - 1 && info->character == leftBracket) {
374  if (matchLeftBracket(textEditor->textCursor().block(), i + 1, 0, leftBracket))
375  createBracketSelection(pos + info->position);
376  } else if (info->position == curPos - 1 && info->character == rightBracket)
377  if (matchRightBracket(textEditor->textCursor().block(), i - 1, 0, rightBracket, true))
378  createBracketSelection(pos + info->position);
379  }
380  }
381  }
382 }

384 bool TextEditor::matchLeftBracket(QTextBlock currentBlock, int index, int numLeftBracket, const char leftBracket)
385 {
386  TextBlockData *data = static_cast<TextBlockData *>(currentBlock.userData());
387  QVector<BracketInfo *> infos = data->brackets();
388  int docPos = currentBlock.position();
389  for (; index < infos.size(); ++index) {
390  BracketInfo *info = infos.at(index);
391  if (info->character == leftBracket) {
392  ++numLeftBracket;
393  continue;
394  }
395  if (info->character == StaticFunctions::rightBracketMatch(leftBracket) && numLeftBracket == 0) {
396  createBracketSelection(docPos + info->position);
397  return true;
398  } else
399  --numLeftBracket;
400  }
401  currentBlock = currentBlock.next();
402  if (currentBlock.isValid())
403  return matchLeftBracket(currentBlock, 0, numLeftBracket, leftBracket);
404  return false;
405 }

698  performCompletion(completionPrefix);

700  cursor.movePosition(QTextCursor::PreviousWord, QTextCursor::KeepAnchor);

724  case EditorTypes::Commit:

726  case EditorTypes::NoEditor:
