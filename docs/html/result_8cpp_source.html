162 void Result::updateResult()
163 {
164  QApplication::setOverrideCursor(Qt::WaitCursor);
165  result = serverConnection->runQuery("SELECT `COLUMN_NAME`"
166  " , IF(`DATA_TYPE` REGEXP '(varchar|char|text|blob).*', '" + StaticFunctions::DelegateTypeNoDelegate() + "'"
167  " , IF(`DATA_TYPE` REGEXP '(float|int).*', '" + StaticFunctions::DelegateTypeNumber() + "'"
168  " , IF(`DATA_TYPE` = 'date', '" + StaticFunctions::DelegateTypeDate() + "'"
169  " , IF(`DATA_TYPE` = 'enum', CONCAT('" + StaticFunctions::DelegateTypeEnum() + "', SUBSTRING(`COLUMN_TYPE`, 5))"
170  " , IF(`DATA_TYPE` REGEXP '(datetime|timestamp).*', '" + StaticFunctions::DelegateTypeDateTime() + "', 0)))))"
171  " , IF(`COLUMN_KEY` = 'PRI', 'PRI', '')"
172  " , `COLUMN_DEFAULT`"
173  " FROM `information_schema`.`COLUMNS` WHERE `TABLE_SCHEMA` = '"
174  + StaticFunctions::unquoteSymbol(p_database) + "' AND `TABLE_NAME` = '"
175  + StaticFunctions::unquoteSymbol(p_table) + "'");
176  result->takeLast(); //Remove the "Affected rows" line.
177  headers->clear();
178  QString fields("'', ");
179  for (int row = 0; row < result->count(); row++) {
180  headers->append(QStringList() << result->at(row).at(0) << result->at(row).at(1)
181  << (result->at(row).at(2).startsWith("PRI") ? ":/images/svg/server-database.svg" : "")
182  << "Left"
183  << result->at(row).at(3));
184  fields += "`" + result->at(row).at(0) + "`, ";
185  }
186  fields = fields.mid(0, fields.length() - 2);
187  resultDTableView->setHeaders(headers);
188  result = serverConnection->runQuery("SELECT " + fields + " FROM " + p_database + "." + p_table + (!p_where.isEmpty() ? " WHERE " + p_where : ""));
189  result->takeLast(); //Remove the "Affected rows" line.
190  if (headers->count() > 0) {
191  resultDTableView->setModelData(result, false);
192  dTitleLabel->setText(tr("Result for: %1.%2").arg(p_database).arg(p_table));
193  resultDTableView->blockSignals(false);
194  } else {
195  emit statusBarMessagePopup(tr("Database or table does not exist."));
196  }
197  QApplication::restoreOverrideCursor();
198 }

200 void Result::itemChangedSlot(QStandardItem *item)
201 {
202  QString statement;
203  if (item->data(80).toString().isEmpty()) {
204  statement = "UPDATE "
205  + p_database + "." + p_table
206  + " SET " + StaticFunctions::quoteSymbol(resultDTableView->getHeaders()->at(item->column()).at(0))
207  + " = " + (resultDTableView->getHeaders()->at(item->column()).at(0) == StaticFunctions::DelegateTypeNumber() ? item->data(Qt::DisplayRole).toString() : "'" + item->data(Qt::DisplayRole).toString() + "'")
208  + " WHERE ";
209  for (int row2 = 0; row2 < headers->count(); row2++)
210  if (headers->at(row2).at(2) == ":/images/svg/server-database.svg")
211  statement += StaticFunctions::quoteSymbol(headers->at(row2).at(0)) + " = '" + (headers->at(row2).at(0) == resultDTableView->getHeaders()->at(item->column()).at(0) ? item->data(Qt::UserRole).toString() : resultDTableView->indexData(item->row(), row2).toString()) + "' AND ";
212  statement = statement.left(statement.length() - 5) + ";";
213  resultDTableView->blockSignals(true);
214  resultDTableView->currentItem()->setBackground(QBrush(QColor("yellow")));
215  resultDTableView->blockSignals(false);
216  } else {
217  statement = "INSERT INTO " + p_database + "." + p_table + " VALUES (";
218  for (int row = 0; row < statements->count(); row++)
219  if (statements->at(row).startsWith(statement))
220  statements->takeAt(row);
221  for (int column = 0; column < resultDTableView->model()->columnCount(); column++)
222  statement += "'" + resultDTableView->indexData(item->row(), column).toString() + "', ";
223  statement = statement.mid(0, statement.length() - 2);
224  statement += ")";
225  }
226  statements->append(statement);
227  discardStatementsAction->setIcon(QIcon::fromTheme("user-trash-full", QIcon(":/images/svg/user-trash-full-4.svg")));
228  discardStatementsAction->setEnabled(!statements->empty());
229  viewChangesAction->setEnabled(!statements->empty());
230 }

232 void Result::deleteCurrentLineActionTriggered()
233 {

235  for (int row2 = 0; row2 < headers->count(); row2++)
236  if (headers->at(row2).at(2) == ":/images/svg/server-database.svg")
237  statement += StaticFunctions::quoteSymbol(headers->at(row2).at(0))
238  + " = '" + resultDTableView->indexData(deleteCurrentLineAction->text().split(": ").at(1).toInt() - 1, row2).toString() + "' AND ";

240  statements->append("DELETE FROM " + p_database + "." + p_table + " WHERE " + statement);

247  resultDTableView->blockSignals(false);
