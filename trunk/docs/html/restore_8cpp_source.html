160 void Restore::getFileInfo(QString file)
161 {
162  if (file.isEmpty())
163  file = fileSelector->getFileName();
164  QFileInfo fileInfo(file);
165  QFile backupFie(file);
166  databaseList.clear();
167  databasesListWidget->clear();
168  tableList.clear();
169  tablesListWidget->clear();
170  backupFie.open(QIODevice::ReadOnly);
171  while (!backupFie.atEnd()) {
172  QString line(backupFie.readLine());
173  if (line.contains(QRegExp("^CREATE.*DATABASE")))
174  addDatabase(line.mid(line.indexOf("`") + 1, line.indexOf("`", line.indexOf("`") + 1) - line.indexOf("`") -1 ));
175  if (line.contains(QRegExp("^CREATE.*TABLE")))
176  addTable(line.mid(line.indexOf("`") + 1, line.indexOf("`", line.indexOf("`") + 1) - line.indexOf("`") -1 ));
177  if (line.contains(QRegExp("^CREATE.*VIEW")))
178  addView(line.mid(line.indexOf("`") + 1, line.indexOf("`", line.indexOf("`") + 1) - line.indexOf("`") -1 ));
179  }
180  schemasLabel->setText(QString("%1").arg(databaseList.count()));
181  fileLabel->setText(fileSelector->getFileName());
182  tablesLabel->setText(QString("%1").arg(tableList.count()));
183  dataLengthLabel->setText("~" + StaticFunctions::bytesConvertor(QString("%1").arg(fileInfo.size())));
184 }
