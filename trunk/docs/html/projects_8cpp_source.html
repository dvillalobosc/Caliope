257 QHash<QString, QString> Projects::getProjectFunctionListComplete()
258 {
259  QHash<QString, QString> data;
260  unsigned int lineNumber = 0;
261  QString param1;
262  QString param3;
263  QString param4;
264  QString previousLine1;
265  QString previousLine2;
266  QString previousLine3;
267  QString previousLine4;
268  QString previousLine5;
269  foreach (QString fileName, getProjectFileList()) {
270  QFile file(fileName);
271  if(file.open(QFile::ReadOnly | QFile::Text)) {
272  QTextStream in(&file);
273  in.setCodec("UTF-8");
274  while(!in.atEnd()) {
275  ++lineNumber;
276  QString line = in.readLine();
277  previousLine5 = previousLine4;
278  previousLine4 = previousLine3;
279  previousLine3 = previousLine2;
280  previousLine2 = previousLine1;
281  previousLine1 = line;
282  if (line.contains(QRegExp("^function.*\\(.*\\) \\{$", Qt::CaseInsensitive))) {
283  if (previousLine5.startsWith(" * @summary:"))
284  param4 = previousLine5.mid(12);
285  if (previousLine4.startsWith(" * @param:"))
286  param3 = previousLine4.mid(10);
287  if (previousLine3.startsWith(" * @return:"))
288  param1 = previousLine3.mid(11);
289  data.insert(line.mid(9, line.indexOf("(") - 9), QString("%1 <u>%2</u>(<b>%3</b>) <br />%4").arg(param1.trimmed()).arg(line.mid(9, line.indexOf("(") - 9)).arg(param3.trimmed()).arg(param4.trimmed()));
290  }
291  param1 = QString();
292  param3 = QString();
293  param4 = QString();
294  }
295  file.close();
296  lineNumber = 0;
297  }
298  }
299  return data;
300 }
