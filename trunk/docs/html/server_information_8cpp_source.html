369 QString ServerInformation::serverGraphsDataTxtMariaDB()

371  QString out;

373  //Bytes sent

381  statementServerGraphs = QString("SELECT FORMAT(%1, 2) AS `T`, FORMAT(%2, 2) AS `T-1`, FORMAT(%3, 2) AS `T-2`, FORMAT(%4, 2) AS `T-3`, FORMAT(%5, 2) AS `T-4`, FORMAT(%6, 2) AS `T-5`")

395  statementServerGraphs = QString("SELECT FORMAT(%1, 2) AS `T`, FORMAT(%2, 2) AS `T-1`, FORMAT(%3, 2) AS `T-2`, FORMAT(%4, 2) AS `T-3`, FORMAT(%5, 2) AS `T-4`, FORMAT(%6, 2) AS `T-5`")
396  .arg(tBytesReceived0 / 1024).arg(tBytesReceived1).arg(tBytesReceived2).arg(tBytesReceived3).arg(tBytesReceived4).arg(tBytesReceived5);
397  out += "\n\n" + tr("Data received in MB.");

404  "'OPEN_TABLES', 'COM_DELETE', 'COM_INSERT', 'COM_SELECT'"
