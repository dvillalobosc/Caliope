170 void ConnectDialog::fillLineEdits(const QString &text)
171 {
172  QStringList params = StaticFunctions::explodeConnectionString(text);
173  //0 - Connection name
174  //1 - User
175  //2 - Host
176  //3 - Port
177  //4 - Database
178  //5 - Conexion count
179  //6 - Collation
180  //7 - UseSSL
181  //8 - KeyFile
182  //9 - CertFile
183  //10 - Password
184  comboConnectionType->setCurrentIndex(comboConnectionType->findText(params.at(0)));
185  lineEditUser->setText(params.at(1));
186  lineEditServer->setText(params.at(2));
187  spinBoxPort->setValue(params.at(3).toUInt());
188  lineEditDatabase->setText(params.at(4));
189  if (storePasswords->isChecked())
190  lineEditPassword->setText(StaticFunctions::password(params.at(10)));
191  setDBMS();
192  count = params.at(5).toInt();
193  lineEditCollation->setText(params.at(6));
194  useASSLConnection->setCheckState(params.at(7) == "1" ? Qt::Checked : Qt::Unchecked);
195  fileSelectorClientKey->setFileName(params.at(8));
196  fileSelectorClientCert->setFileName(params.at(9));
197 }

208 void ConnectDialog::databasesMenuSlot()
209 {
210  QApplication::setOverrideCursor(Qt::WaitCursor);
211  databasesMenu->clear();
212  if (!serverConnection->isOpened()) {
213  serverConnection->setUserName(lineEditUser->text());
214  serverConnection->setHostName(lineEditServer->text());
215  serverConnection->setPassword(lineEditPassword->text());
216  serverConnection->setDatabase(StaticFunctions::DBMSDefaultDatabase());
217  serverConnection->setPort(spinBoxPort->value());
218  serverConnection->setCharacterSet(lineEditCollation->text().split("|").at(0));
219  serverConnection->setCollation(lineEditCollation->text().split("|").at(1));
220  serverConnection->setUseSSL(useASSLConnection->isChecked());
221  serverConnection->setKeyFile(fileSelectorClientKey->getFileName());
222  serverConnection->setCertFile(fileSelectorClientCert->getFileName());
223  serverConnection->open();
224  connectionPerformed = true;
225  }
226  foreach (QString database, serverConnection->getDatabases()) {
227  QAction *action = databasesMenu->addAction(database);
228  action->setCheckable(true);
229  action->setChecked(action->text() == lineEditDatabase->text());
230  connect(action, SIGNAL(triggered()), databasesMapper, SLOT(map()));
231  databasesMapper->setMapping(action, database);
232  }
233  QApplication::restoreOverrideCursor();
234 }

291 void ConnectDialog::actionRoleSlot(QAbstractButton *button)

295  serverConnection->setUserName(lineEditUser->text());
296  serverConnection->setHostName(lineEditServer->text());
297  serverConnection->setPassword(lineEditPassword->text());
298  serverConnection->setDatabase(StaticFunctions::DBMSDefaultDatabase());
299  serverConnection->setPort(spinBoxPort->value());
300  serverConnection->setCharacterSet(lineEditCollation->text().split("|").at(0));
301  serverConnection->setCollation(lineEditCollation->text().split("|").at(1));

365  .arg(comboConnectionType->currentText())
366  .arg(lineEditUser->text())
367  .arg(lineEditServer->text())
368  .arg(spinBoxPort->value())
369  .arg(lineEditDatabase->text())
370  .arg(++count)
371  .arg(lineEditCollation->text())
372  .arg(useASSLConnection->isChecked())
373  .arg(fileSelectorClientKey->getFileName().isEmpty() ? "-" : fileSelectorClientKey->getFileName())
374  .arg(fileSelectorClientCert->getFileName().isEmpty() ? "-" : fileSelectorClientCert->getFileName())
375  .arg(storePasswords->isChecked() ? StaticFunctions::password(lineEditPassword->text(), true) : "")
376  );
377  settings.setValue("StorePassword", storePasswords->isChecked());
378  settings.setValue("SortConnectionList", sortConnectionList->isChecked());
379  return listValues;
