152 void ConnectDialog::databasesMenuSlot()
153 {
154  QApplication::setOverrideCursor(Qt::WaitCursor);
155  databasesMenu->clear();
156  if (!serverConnection->isOpened()) {
157  serverConnection->setUserName(lineEditUser->text());
158  serverConnection->setHostName(lineEditServer->text());
159  serverConnection->setPassword(lineEditPassword->text());
160  serverConnection->setDatabase(StaticFunctions::DBMSDefaultDatabase());
161  serverConnection->setPort(spinBoxPort->value());
162  serverConnection->open();
163  connectionPerformed = true;
164  }
165  foreach (QString database, serverConnection->getDatabases()) {
166  QAction *action = databasesMenu->addAction(database);
167  action->setCheckable(true);
168  action->setChecked(action->text() == lineEditDatabase->text());
169  connect(action, SIGNAL(triggered()), databasesMapper, SLOT(map()));
170  databasesMapper->setMapping(action, database);
171  }
172  QApplication::restoreOverrideCursor();
173 }
