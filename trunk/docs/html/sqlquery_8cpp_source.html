427 void SQLQuery::execureStatement(QString statement)
428 {
429  if (!statement.isEmpty()) {
430  if (statement.contains(QRegExp("^(ALTER|CREATE|DELETE|DROP|GRANT|LOAD|RENAME|TRUNCATE|UPDATE)", Qt::CaseInsensitive))
431  && safeStatementsAction->isChecked()) {
432  QString message(tr("Could not execute statement on safe mode."));
433  QMessageBox::warning(this, tr("Safe mode"), message);
434  resutlEditor->setPlainText(message + "\n" + statement);
435  return;
436  }
437  if (logStatements) //Use a variable here because is faster
438  serverConnection->logStatement(statement);
439  serverConnection->executedQueries->append(statement);
440  if (settings.value("SaveQueriesBeforeExecution", true).toBool())
441  settings.setValue("LastQuery-" + qApp->property("ConnectionName").toString() + "-" + windowTitle(), scriptEditor->textEditor->toPlainText().trimmed());
442  if (!concatenateOutputAction->isChecked())
443  resutlEditor->clear();
444  if (radioT->isChecked())
445  resutlEditor->setPlainText(serverConnection->outputAsTable(statement, false, exportAction->isChecked()
446  , showNewLinesAction->isChecked(), splitAction->isChecked())
447  , concatenateOutputAction->isChecked());
448  if (radioV->isChecked())
449  resutlEditor->setPlainText(serverConnection->outputAsV(statement, false, exportAction->isChecked()
450  , showNewLinesAction->isChecked(), splitAction->isChecked())
451  , concatenateOutputAction->isChecked());
452  if (radioVV->isChecked())
453  resutlEditor->setPlainText(serverConnection->outputAsVV(statement, exportAction->isChecked()
454  , showNewLinesAction->isChecked(), splitAction->isChecked())
455  , concatenateOutputAction->isChecked());
456  if (radioVVV->isChecked())
457  resutlEditor->setPlainText(serverConnection->outputAsTable(statement, true, exportAction->isChecked()
458  , showNewLinesAction->isChecked(), splitAction->isChecked())
459  , concatenateOutputAction->isChecked());
460  if (radioHTML->isChecked())
461  resutlEditor->setPlainText(serverConnection->outputAsHTML(statement, exportAction->isChecked()
462  , showNewLinesAction->isChecked(), splitAction->isChecked())
463  , concatenateOutputAction->isChecked());
464  if (radioTXT->isChecked())
465  resutlEditor->setPlainText(serverConnection->outputAsTable(statement, false, exportAction->isChecked()
466  , showNewLinesAction->isChecked(), splitAction->isChecked())
467  , concatenateOutputAction->isChecked());
468  if (radioXML->isChecked())
469  resutlEditor->setPlainText(serverConnection->outputAsXML(statement, exportAction->isChecked()
470  , showNewLinesAction->isChecked(), splitAction->isChecked())
471  , concatenateOutputAction->isChecked());
472  if (radioG->isChecked())
473  resutlEditor->setPlainText(serverConnection->outputAsG(statement, exportAction->isChecked()
474  , showNewLinesAction->isChecked(), splitAction->isChecked())
475  , concatenateOutputAction->isChecked());
476  if (radioPDF->isChecked()) {
477  resutlEditor->setPlainText(serverConnection->outputAsTable(statement, false, false, showNewLinesAction->isChecked(), splitAction->isChecked()));
478  QApplication::setOverrideCursor(Qt::WaitCursor);
479  QFileDialog fileDialog;
480  fileDialog.setDirectory(QDir::home());
481  QString file = fileDialog.getSaveFileName(this, tr("Save to Pdf"), settings.value("LastFilePdf", QDir::home().absolutePath()).toString(), tr("Pdf & Ps files (*.pdf *.ps)"));
482  settings.setValue("LastFilePdf", fileDialog.directory().filePath(file));
483  QPrinter printer(QPrinter::HighResolution);
484  printer.setOutputFileName(file);
485  printer.setOutputFormat(file.endsWith(".pdf") ? QPrinter::PdfFormat : QPrinter::NativeFormat);
486  resutlEditor->document()->print(&printer);
