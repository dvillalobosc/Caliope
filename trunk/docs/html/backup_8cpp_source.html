39 Backup::Backup(DBMS *serverConnection)
40 {
41  this->serverConnection = serverConnection;
42  setWindowIcon(QIcon(":/images/svg/server-database.svg"));
43  QWidget *widMain = new QWidget;
44  QVBoxLayout *mainVLayout = new QVBoxLayout;
45  QHBoxLayout *fileLayout = new QHBoxLayout;
46  QHBoxLayout *databaseLayout = new QHBoxLayout;
47  mainVLayout->setContentsMargins(3, 0, 3, 0);
48  dTitleLabel = new DTitleLabel;
49  mainVLayout->addWidget(dTitleLabel);
50  dMessageLabel = new DMessageLabel(MessageTypes::Information);
51  mainVLayout->addWidget(dMessageLabel);
52  gridChecks = new QGridLayout;
53  databasesListWidget = new QListWidget;
54  databasesListWidget->setMaximumWidth(200);
55  connect(databasesListWidget, SIGNAL(clicked(QModelIndex)), this, SLOT(disbaleBackItIp()));
56  foreach (QString database, serverConnection->getDatabases()) {
57  QListWidgetItem *item = new QListWidgetItem(QIcon(":/images/svg/server-database.svg"), database);
58  item->setFlags(Qt::ItemIsUserCheckable | Qt::ItemIsEnabled | Qt::ItemIsSelectable);
59  item->setCheckState(Qt::Unchecked);
60  databasesListWidget->addItem(item);
61  }
62  foreach (QString option, settings.value("Backup/Databases", QStringList()).toStringList())
63  if (!option.isEmpty())
64  for (int counter = 0; counter < databasesListWidget->count(); counter++)
65  if (databasesListWidget->item(counter)->text() == option)
66  databasesListWidget->item(counter)->setCheckState(Qt::Checked);
67  switch(serverConnection->getDBMSType()) {
68  case StaticFunctions::MySQL:
69  case StaticFunctions::MariaDB:
70  createCheckBoxesMariaDB();
71  foreach (QString option, settings.value("Backup/Options", QStringList()).toStringList())
72  if (!option.isEmpty())
73  for (int counter = 0; counter < gridChecks->count(); ++counter) {
74  item = qobject_cast<QCheckBox *>(gridChecks->itemAt(counter)->widget());
75  if (item->text() == option)
76  item->setChecked(true);
77  }
78  break;
79  case StaticFunctions::Undefined:
80  default:
81  QMessageBox::information(this, tr("Cannot backup this kind of database"), tr("Cannot backup this kind of database"));
82  break;
83  }
84  databaseLayout->addWidget(databasesListWidget);
85  databaseLayout->addLayout(gridChecks);
86  databaseLayout->addStretch();
87  mainVLayout->addLayout(databaseLayout);
88  fileSelector = new FileSelector(FileSelectorContexts::BackupFile);
89  fileLayout->addWidget(fileSelector);
90  connect(fileSelector, SIGNAL(changed()), this, SLOT(disbaleBackItIp()));
91  backItUpPushButton = new QPushButton;
92  backItUpPushButton->setEnabled(false);
93  connect(backItUpPushButton, SIGNAL(clicked()), this, SLOT(backItUpSlot()));
94  disbaleBackItIp();
95  fileLayout->addWidget(backItUpPushButton);
96  mainVLayout->addLayout(fileLayout);
97  widMain->setLayout(mainVLayout);
98  retranslateUi();
99  setWidget(widMain);
100 }

134 void Backup::backItUpSlot()
135 {
136  QStringList options;
137  proc = new QProcess;
138  connect(proc, SIGNAL(readyReadStandardError()), this, SLOT(readyReadStandardErrorSlot()));
139  connect(proc, SIGNAL(finished(int,QProcess::ExitStatus)), this, SLOT(finishedSlot(int)));
140  QStringList arguments;
141  arguments << "--user=" + serverConnection->getUserName();
142  arguments << "--host=" + serverConnection->getHostName();
143  arguments << "--password=" + serverConnection->getPassword();
144 // arguments << "--port=" + serverConnection->getPort();
145  arguments << "--default-character-set=utf8";
146  // arguments << "--port=" + serverConnection->getPort();
147  arguments << "--result-file=" + fileSelector->getFileName();
148  for (int counter = 0; counter < gridChecks->count(); ++counter) {
149  item = qobject_cast<QCheckBox *>(gridChecks->itemAt(counter)->widget());
150  if (item->isChecked()) {
151  arguments << item->text();
152  options << item->text();
153  }
154  }
155  settings.setValue("Backup/Options", options);
156  options.clear();
157 
158  if (allDatabases->isChecked()) {
159  arguments << "--all-databases";
160  } else {
161  arguments << "--databases";
162 // foreach(QListWidgetItem *item, databasesListWidget->selectedItems())
163 // arguments << item->text();
164  for (int counter = 0; counter < databasesListWidget->count(); counter++)
165  if (databasesListWidget->item(counter)->checkState() == Qt::Checked)
166  arguments << databasesListWidget->item(counter)->text();
167  }
168  settings.setValue("Backup/Databases", options);
169  QApplication::setOverrideCursor(Qt::WaitCursor);
170  proc->start("mysqldump", arguments);
171  proc->waitForFinished();
172  QApplication::restoreOverrideCursor();
173 }
