91 void NetworkSettings::setProxySettings()
92 {
93  QSettings settings;
94  QNetworkProxy proxy;
95  if (settings.value("Network/UseProxy", false).toBool()) {
96  proxy.setType(QNetworkProxy::HttpProxy);
97  proxy.setHostName(settings.value("Network/ProxyAddress", "").toString());
98  proxy.setPort(settings.value("Network/Port", 0).toInt());
99  if (!settings.value("Network/User", "").toString().isEmpty())
100  proxy.setUser(settings.value("Network/User", "").toString());
101  if (!settings.value("Network/Password", "").toString().isEmpty())
102  proxy.setPassword(StaticFunctions::password(settings.value("Network/Password", "").toString()));
103  } else {
104  proxy.setType(QNetworkProxy::NoProxy);
105  }
106  QNetworkProxy::setApplicationProxy(proxy);
107 }
