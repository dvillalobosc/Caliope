253 void Catalogs::tablesItemChangedSlot(QStandardItem *item)
254 {
255  bool done = false;
256  if (!readOnly) {
257  switch(item->column()) {
258  case 0:
259  done = serverConnection->table(item->data(Qt::UserRole).toString())->renameTable(item->data(Qt::DisplayRole).toString());
260  break;
261  case 1:
262  done = serverConnection->table(tablesDTableView->indexData(item->row(), 0, Qt::UserRole).toString())
263  ->changeEngine(item->data(Qt::DisplayRole).toString());
264  break;
265  case 14:
266  done = serverConnection->table(tablesDTableView->indexData(item->row(), 0, Qt::UserRole).toString())
267  ->changeCollation(item->data(Qt::DisplayRole).toString());
268  break;
269  case 17:
270  done = serverConnection->table(tablesDTableView->indexData(item->row(), 0, Qt::UserRole).toString())
271  ->changeComment(item->data(Qt::DisplayRole).toString());
272  break;
273  default:
274  emit statusBarMessagePopup(tr("The column %1 cannot be modified.").arg(tablesDTableView->getHeaders()->at(item->column()).at(0)));
275  break;
276  }
277  if (done)
278  emit statusBarMessagePopup(tr("Changes applied."));
279  tablesInfo();
280  } else {
281  emit statusBarMessage(tr("Tables in Read-Only mode, no changes will be applied."), QSystemTrayIcon::Information);
282  }
283 }

297 QString Catalogs::valueToPopulate(QString dataType, unsigned int length)
298 {
299  if (dataType.contains(QRegExp("(varchar|longtext|char|text|mediumblob|longblob|mediumtext|blob|tinyblob|binary|varbinary|tinytext)")))
300  return "'" + StaticFunctions::randomString(length) + "'";
301  if (dataType.contains(QRegExp("(datetime|timestamp)")))
302  return "NOW()";
303  if (dataType.contains(QRegExp("(date)")))
304  return "DATE(NOW())";
305  if (dataType.contains(QRegExp("(year)")))
306  return "YEAR(NOW())";
307  if (dataType.contains(QRegExp("(time)")))
308  return "TIME(NOW())";
309  if (dataType.contains(QRegExp("(set|enum)")))
310  return "1";
311  if (dataType.contains(QRegExp("(bigint|int|decimal|double|smallint|mediumint|tinyint|float|bit)")))
312  return QString("%1").arg((qrand() % length) + 1);
313  return QString();
314 }
