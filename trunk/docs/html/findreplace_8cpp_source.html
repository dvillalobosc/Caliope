102 bool FindReplace::find(QString text, QTextDocument::FindFlags direction)
103 {
104  if (text.isEmpty())
105  text = findLineEdit->text();
106  QTextDocument::FindFlags flags;
107  if (direction)
108  flags |= QTextDocument::FindBackward;
109  if (caseSensitiveAction->isChecked())
110  flags |= QTextDocument::FindCaseSensitively;
111  if (entireWordAction->isChecked())
112  flags |= QTextDocument::FindWholeWords;
113  if (regExpAction->isChecked()) {
114  QTextCursor cursor = plainTextEdit->textEditor->document()->find(QRegExp(text, Qt::CaseInsensitive), plainTextEdit->textEditor->textCursor(), flags);
115  plainTextEdit->textEditor->setTextCursor(cursor);
116  if (cursor.hasSelection())
117  return true;
118  else
119  return false;
120  } else {
121  if (plainTextEdit->textEditor->find(text, flags)) {
122  return true;
123  } else {
124  return false;
125  }
126  }
127 }
