118 QString TableMaintenance::tableList()

217 void TableMaintenance::itemActivatedSlot(QTreeWidgetItem *item, int column)
218 {
219  if (item->type() == ItemTypes::Database) {
220  QApplication::setOverrideCursor(Qt::WaitCursor);
221  foreach (QTreeWidgetItem *tableItem, tables)
222  if (tableItem->parent() == item)
223  tableItem->setCheckState(column, item->checkState(column));
224  if (item->childCount() == 0)
225  foreach (QString table, serverConnection->database(item->text(0))->getTables()) {
226  QTreeWidgetItem *itemChild = new QTreeWidgetItem(item, QStringList(item->text(0) + "." + table), ItemTypes::Table);
227  itemChild->setFlags(Qt::ItemIsUserCheckable | Qt::ItemIsEnabled | Qt::ItemIsSelectable);
228  itemChild->setIcon(0, QIcon(":/images/svg/table.svg"));
229  itemChild->setCheckState(0, Qt::Unchecked);
230  tables.append(itemChild);
231  }
232  QApplication::restoreOverrideCursor();
233  }
234 }
