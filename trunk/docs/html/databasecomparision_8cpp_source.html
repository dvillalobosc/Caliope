185 void DatabaseComparision::comparision(bool primary)
186 {
187  QApplication::setOverrideCursor(Qt::WaitCursor);
188  QList<QStringList> *rows = new QList<QStringList>();
189  rows->append(QStringList() << tr("Server") << tr("Database") << tr("Table") << tr("Row count") << tr("Data length") << tr("Checksum")
190  << tr("Row count validation")
191  << tr("Data length validation")
192  << tr("Checksum validation")
193  );
194  QString database;
195  QString table;
196  QList<QTreeWidgetItem *> tablesToCompare;
197  int unsigned itemProcessed = 0;
198  unsigned long rowCount1 = 0;
199  unsigned long rowCount2 = 0;
200  unsigned long dataLength1 = 0;
201  unsigned long dataLength2 = 0;
202  unsigned long checksum1 = 0;
203  unsigned long checksum2 = 0;
204 
205  if (primary)
206  tablesToCompare = tables;
207  else
208  tablesToCompare = secondaryTables;
209 
210  foreach (QTreeWidgetItem *item, tables) {
211  if (item->checkState(0) == Qt::Checked && item->parent()) {
212  database = item->text(0).split(".").at(0);
213  table = item->text(0).split(".").at(1);
214  rowCount1 = serverConnection->table(table, database)->getRowCount();
215  rowCount2 = secondaryServerConnection->table(table, database)->getRowCount();
216  dataLength1 = serverConnection->table(table, database)->getDataLength();
217  dataLength2 = secondaryServerConnection->table(table, database)->getDataLength();
218  checksum1 = serverConnection->table(table, database)->getChecksum();
219  checksum2 = secondaryServerConnection->table(table, database)->getChecksum();
220  rows->append(QStringList() << serverConnection->getHostName()
221  << database
222  << table
223  << QString("%1").arg(rowCount1)
224  << QString("%1").arg(dataLength1)
225  << QString("%1").arg(checksum1)
226  << (primary ? (rowCount1 != rowCount2 ? "*" : "") : "")
227  << (primary ? (dataLength1 != dataLength2 ? "*" : "") : "")
228  << (primary ? (checksum1 != checksum2 ? "*" : "") : "")
229  );
230  rows->append(QStringList() << secondaryServerConnection->getHostName()
231  << database
232  << table
233  << QString("%1").arg(rowCount2)
234  << QString("%1").arg(dataLength2)
235  << QString("%1").arg(checksum2)
236  << (primary ? "" : (rowCount1 != rowCount2 ? "*" : ""))
237  << (primary ? "" : (dataLength1 != dataLength2 ? "*" : ""))
238  << (primary ? "" : (checksum1 != checksum2 ? "*" : ""))
239  );
240  rows->append(QStringList() << "***"
241  << "***"
242  << "***"
243  << "***"
244  << "***"
245  << "***"
246  << "***"
247  << "***"
248  << "***"
249  );
250  }
251  emit statusBarProgressMessage(item->text(0), 0, (++itemProcessed * 100 / tables.count()));
252  }
253  //Remove the last ***
254  rows->takeLast();
255 
256  int row = 0;
257  int row2 = 0;
258  QList<int> maxWidthList;
259  QString output;
260 
261  //Find the maximum values for the width of the columns
262  for (row = 0; row < rows->at(0).count(); row++)
263  maxWidthList.append(1);
264  for (row = 0; row < rows->count(); row++)
265  for (row2 = 0; row2 < rows->at(row).count(); row2++)
266  if (QString(rows->at(row).at(row2)).length() > maxWidthList.at(row2))
267  maxWidthList.replace(row2, QString(rows->at(row).at(row2)).length());
268 
269  //Prints the headres and theirs borders
270  output += "+";
271  for (row = 0; row < rows->at(0).count(); row++)
272  output += QString('-').repeated(maxWidthList.at(row) + 2) + "+";
273  output += "\n|";
274  for (row = 0; row < rows->at(0).count(); row++)
275  output += " " + rows->at(0).at(row).leftJustified(maxWidthList.at(row), ' ') + " |";
276  output += "\n+";
277  for (row = 0; row < rows->at(0).count(); row++)
278  output += QString('-').repeated(maxWidthList.at(row) + 2) + "+";
279 
280  //Print the data
281  for (row = 1; row < rows->count(); row++) {
282  //output += "\n|" + ((row % 2 != 0) ? highlightError(rows, row) : "");
283  output += "\n|";
284  for (row2 = 0; row2 < rows->at(row).count(); row2++)
285  if (rows->at(row).at(row2) == "***")
286  output += QString('-').repeated(maxWidthList.at(row2) + 2) + "|";
287  else
288  output += " " + rows->at(row).at(row2).leftJustified(maxWidthList.at(row2), ' ') + " |";
289  }
290 
291  //Print the botton border
292  output += "\n+";
293  for (row = 0; row < rows->at(0).count(); row++)
294  output += QString('-').repeated(maxWidthList.at(row) + 2) + "+";
295 
296  resutlEditor->setPlainText(output);
297  QApplication::restoreOverrideCursor();
298 }

305 void DatabaseComparision::itemActivatedSlot(QTreeWidgetItem *item, int column)
306 {
307  if (item->type() == ItemTypes::Database) {
308  QApplication::setOverrideCursor(Qt::WaitCursor);
309  foreach (QTreeWidgetItem *tableItem, tables)
310  if (tableItem->parent() == item)
311  tableItem->setCheckState(column, item->checkState(column));
312  if (item->childCount() == 0)
313  foreach (QString table, serverConnection->database(item->text(0))->getLocalTables()) {
314  QTreeWidgetItem *itemChild = new QTreeWidgetItem(item, QStringList(item->text(0) + "." + table), ItemTypes::Table);
315  itemChild->setFlags(Qt::ItemIsUserCheckable | Qt::ItemIsEnabled | Qt::ItemIsSelectable);
316  itemChild->setIcon(0, QIcon(":/images/svg/table.svg"));
317  itemChild->setCheckState(0, Qt::Unchecked);
318  tables.append(itemChild);
