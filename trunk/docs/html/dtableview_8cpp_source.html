53 void DTableView::setHeaders()
54 {
55  if (!this->readOnly) {
56  for (int column = 0; column < headersList->count(); column++) {
57  if (headersList->at(column).at(1) == StaticFunctions::DelegateTypeNumber())
58  setItemDelegateForColumn(column, new DItemDelegate(StaticFunctions::DelegateTypeNumber()));
59  if (headersList->at(column).at(1) == StaticFunctions::DelegateTypeDateTime())
60  setItemDelegateForColumn(column, new DItemDelegate(StaticFunctions::DelegateTypeDateTime()));
61  if (headersList->at(column).at(1) == StaticFunctions::DelegateTypePassword())
62  setItemDelegateForColumn(column, new DItemDelegate(StaticFunctions::DelegateTypePassword()));
63  if (headersList->at(column).at(1) == StaticFunctions::DelegateTypeDate())
64  setItemDelegateForColumn(column, new DItemDelegate(StaticFunctions::DelegateTypeDate()));
65  if (headersList->at(column).at(1).startsWith(StaticFunctions::DelegateTypeEnum()))
66  setItemDelegateForColumn(column, new DItemDelegate(headersList->at(column).at(1)));
67  }
68  } else {
69  for (int column = 0; column < headersList->count(); column++)
70  setItemDelegateForColumn(column, new QItemDelegate());
71  }
72 }

79 void DTableView::setModelData(QList<QStringList> *modelData, bool readOnly, unsigned int orderColumn)
80 {
81  QApplication::setOverrideCursor(Qt::WaitCursor);
82  this->readOnly = readOnly;
83  setHeaders();
84  emit loadStarted(tr("Loading records..."), 0, 0);
85  itemModel = new QStandardItemModel(modelData->count(), headersList->count());
86  itemModel->blockSignals(true);
87  connect(itemModel, SIGNAL(itemChanged(QStandardItem*)), this, SLOT(itemChangedSlot(QStandardItem*)));
88  for (int row = 0; row < modelData->count(); row++) {
89  for (int column = 0; column < headersList->count(); column++) {
90  itemModel->setHorizontalHeaderItem(column, new QStandardItem(QIcon(headersList->at(column).at(2)), headersList->at(column).at(0)));
91  QStandardItem *item;
92  if (column == 0)
93  item = new QStandardItem(QIcon(modelData->at(row).at(0)), modelData->at(row).at(column + 1));
94  else
95  item = new QStandardItem(modelData->at(row).at(column + 1));
96  if (headersList->at(column).at(3) == "Right")
97  item->setTextAlignment(Qt::AlignRight | Qt::AlignVCenter);
98  if (headersList->at(column).at(3) == "ToBytes") {
99  item->setText(StaticFunctions::bytesConvertor(item->text()));
100  item->setTextAlignment(Qt::AlignRight | Qt::AlignVCenter);
101  }
102  itemModel->setItem(row, column, item);
103  if (!this->readOnly)
104  itemModel->setData(itemModel->index(row, column, QModelIndex()), modelData->at(row).at(column + 1), Qt::UserRole);
105  }
106  emit loadProgress(tr("Loading records..."), 0, row * 100 / modelData->count());
107  }
108  QItemSelectionModel *oldModel = selectionModel();
109  setModel(itemModel);
110  delete oldModel;
111  resizeColumnsToContents();
112  resizeRowsToContents();
113  setSortingEnabled(true);
114  if (orderColumn == 0)
115  orderColumn = sortingColumn;
116  sortByColumn(orderColumn, sortOrder);
117  itemModel->blockSignals(false);
118  emit loadStarted(tr("Records loaded."), 0, 100);
119  QApplication::restoreOverrideCursor();
120 }
