64 QString CodeSnippets::loadCodeSnippets()
65 {
66  QString out;
67  settings.beginGroup("CodeSnippets");
68  QStringList codeSnippets = settings.allKeys();
69  codeSnippets.removeAt(codeSnippets.indexOf("Snippets"));
70  settings.endGroup();
71  foreach (QString snippet, codeSnippets)
72  out += "###" + snippet + "$$$\n" + getCodeSnippet(snippet) + "\n\n";
73  return out.trimmed();
74 }

98 void CodeSnippets::saveCodeSnippets()
99 {
100  QStringList data;
101  settings.beginGroup("CodeSnippets");
102  settings.group().clear();
103  foreach (QString snippet, codeSnippetsPlainTextEdit->toPlainText().split("###", QString::SkipEmptyParts)) {
104  data = snippet.split("$$$", QString::SkipEmptyParts);
105  settings.setValue(data.at(0), data.at(1).trimmed());
106  }
107  settings.endGroup();
